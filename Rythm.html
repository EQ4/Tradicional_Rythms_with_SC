      *defineProxyPattern 

      ~instrumentPPatt = PatternProxy(Pseq([\simplePlayBuf], inf));
      ~ratePPatt = PatternProxy(Pseq([1], inf));
      ~bufnumPPatt = PatternProxy(Pseq([~dum, ~te], inf));
      ~ampPPatt = PatternProxy(Pseq([0, 0], inf));
      ~durPPatt = PatternProxy(Pseq([1, 1], inf));
